An Array of some Cannons

We've taken hits! Through sheer coincidence our lookout is reporting that they're sailing the exact kind of ship!

The code for their cannon array is below, if their cannons have fired from their starboard side and hit exactly their exact counterparts on our port but in reverse...
i.e. Their 1st Starboard Cannon hit and destroyed our 6th Port cannon, what cannons do we have left to fire with?

        public class Cannons {
            public static void main (String[] args) {
                int[] cannonStarboard = { 1 , 1 , 1 , 0 , 0 , 0 };
                int[] cannonPort = { 1 , 1 , 0 , 0 , 1 , 1 };
                String cannonStatus;
        
                System.out.println("Cannon status report! Let's sink that Poulterheist!");
                System.out.print("Starboard Cannons: ");
                for (int i = 0 ; i < cannonStarboard.length ; ++i) {
                    if (cannonStarboard[i] == 1) {
                        cannonStatus = "Ready!";
                    }
                    else {
                        cannonStatus = "Empty!";
                    }
                    System.out.print((i+1) + ": " + cannonStatus + " ");
                }
                System.out.println("");
                System.out.print("Port Cannons: ");
                for (int i = 0 ; i < cannonPort.length ; ++i) {
                    System.out.print(cannonPort[i] + " ");
                    if (cannonPort[i] == 1) {
                        cannonStatus = "Ready!";
                    }
                    else {
                        cannonStatus = "Empty!";
                    }
                    System.out.print((i+1) + ": " + cannonStatus + " ");
                }
                    System.out.println("");
            }
        }
