An Array of Cannons

BRIGANDS. Off the port side and they’re with the wind!
We’ve got to bring our cannons to bear, make sure our gunners belowdecks know which set to load and ready!

Below is the code for our cannon arrays on both the starboard and port sides, how many cannons do we have at the ready?

        public class Cannons {
            public static void main (String[] args) {
                int[] cannonStarboard = { 0 , 1 , 0 , 1 , 1 , 0 };
                int[] cannonPort = { 0 , 1 , 0 , 0 , 1 , 1 };
                String cannonStatus;
        
                System.out.println("Cannon status report!");
                System.out.print("Starboard Cannons: ");
                for (int i = 0 ; i < cannonStarboard.length ; ++i) {
                    if (cannonStarboard[i] == 1) {
                        cannonStatus = "Ready!";
                    }
                    else {
                        cannonStatus = "Empty!";
                    }
                    System.out.print((i+1) + ": " + cannonStatus + " ");
                }
                System.out.println("");
                System.out.print("Port Cannons: ");
                for (int i = 0 ; i < cannonPort.length ; ++i) {
                    System.out.print(cannonPort[i] + " ");
                    if (cannonPort[i] == 1) {
                        cannonStatus = "Ready!";
                    }
                    else {
                        cannonStatus = "Empty!";
                    }
                    System.out.print((i+1) + ": " + cannonStatus + " ");
                }
                    System.out.println("");
            }
        }
